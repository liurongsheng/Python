## 数据库写入

MySQL 写入基础版本
```python
import pymysql
class MySQLPipeline(object):
    def __init__(self):
        pass

    def process_item(self, item, spider):
        db = pymysql.connect(
            host="127.0.0.1",
            user='root',
            password='admin123',
            database='pymysql',
            port=3306
        )
        cursor = db.cursor()
        sql = """
            insert into user(
                id,title,href
            ) 
            values(null,%s,%s);
        """
        cursor.execute(sql,[item['title'],item['href']])
        db.commit()
        cursor.close()
```

MongoDB 写入基础版本
```python
import pymongo
from y2w1.items import Y2W1Item
class MongoDBPipeline(object):
    def __init__(self):
        client = pymongo.MongoClient("localhost",27017)
        db = client["y2w1"]
        self.Y2W1Item = db["Y2W1Item"]
        
    def process_item(self, item, spider):
        if isinstance(item, Y2W1Item):
            try:
                self.Y2W1Item.insert(dict(item))
            except Exception:
                pass
        return item
```
